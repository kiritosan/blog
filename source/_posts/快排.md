let arr = [1,2,3,4,23,2,2];

function quickSort(arr, left, right) {
  if (left >= right) {
    return arr;
  }
  let l = left,
      r = right,
      pivot = arr[left];
  function partition(arr) { // 两件事:1.根据pivot分成两块2.返回pivot索引
    while (l < r) {
      while (l < r && arr[r] >= pivot) { //右边大于等于就忽略
        r--;
      }
      while (l < r && arr[l] <= pivot) { //左边小于等于就忽略
        l++;
      }
      if(l < r) {
        arr[left] = arr[r]; // 不用交换 直接移动
        arr[r] = arr[l];
      }
      
      arr[left] = arr[l];
      arr[l] = pivot;
    }
    return l;
  }



  let pivotIndex = partition(arr);
  quickSort(arr, left, pivotIndex);
  quickSort(arr, pivotIndex + 1, right);

  return arr;
}

quickSort(arr, 0, arr.length);
