function heapSort(arr) {
    let n = arr.length;
    if (n <= 1) {
        return arr;
    }
    // build a heap 从后往前 从上到下
    for (let i = Math.floor(n/2); i >= 0; i--) {
        heapify(arr, i, n);
    }
    // sort
    // 循环来缩短堆的长度
    for (let i = 0; i < n; i++) {
        // 第一个和最后一个交换
        swap(arr, 0, n - 1 - i);
        // 只需heapify第一个元素 堆长度减一
        heapify(arr, 0, n - 2 - i);
    }
    return arr;
}

function swap(arr, a, b) {
    [arr[a], arr[b]] = [arr[b], arr[a]];
}
// ！！这里的size不是堆的长度，而是堆最后一个元素的索引 所以l<<size
function heapify(arr, i, size) {
    let l = 2*i+1;
    let r = 2*i+2;
    let largest = i;
    if (l <= size && arr[l] > arr[i]) {
        largest = l;
    }
    if (r <= size && arr[r] > arr[i]) {
        largest = r;
    }
    if (largest !== i) {
        swap(arr, i, largest);
        heapify(arr, largest, size);
    }
}

let arr = [4,5,3,4,5]
heapSort(arr)

function heapSort(arr) {
    let n = arr.length
    if (n <= 1) return arr
    // 从后往前 build heap
    for(let i = Math.floor(n/2); i >= 0; i--) {
        maxHeapify(arr, i, n)
    }
    for (let i = 0; i < n; i++) {
        swap(arr, 0, n - 1 - i)
        maxHeapify(arr, 0, n - 1 - i)
    }
    return arr
}

function swap(arr, a, b) {
    [arr[a],arr[b]] = [arr[b],arr[a]]
}
// size為堆的自然長度 所以l<size为条件
function maxHeapify(arr, i, size) {
    let l = 2 * i + 1, r = 2 * i + 2
    // 假定largest是i,但這是不穩定的
    let largest = i
    if(l<size && arr[l]>arr[largest]){
        largest = l
    }
    if(r<size && arr[r]>arr[largest]){
        largest = r
    }
    if(largest !== i){
        swap(arr, i, largest)
        // 交換後largest又不穩定了
        maxHeapify(arr, largest, size)
    }


}
